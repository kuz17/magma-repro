class SoMDataset(Dataset):
    def __init__(self, jsonl_path, renders_dir):
        self.renders_dir = Path(renders_dir)
        self.samples = []
        with open(jsonl_path) as f:
            for line in f:
                line = line.strip()
                if not line: continue
                entry = self._parse(json.loads(line))
                if entry: self.samples.append(entry)
        print(f'Dataset: {len(self.samples):,} valid samples')

    def _parse(self, sample):
        convs = sample.get('conversations', [])
        if len(convs) < 2: return None
        som_path = None
        if sample.get('som_image'):
            p = Path(sample['som_image'])
            if p.exists(): som_path = p
        if som_path is None:
            stem = Path(sample['image']).stem
            p = self.renders_dir / f'{stem}_som.png'
            if p.exists(): som_path = p
        if som_path is None: return None
        user_text = re.sub(r'^<image>\n?', '', convs[0]['value']).strip()
        asst_text = convs[1]['value']
        if not any(kw in asst_text for kw in ('Coordinate:', 'coordinate:')):
            return None
        return {'som_path': som_path, 'user': user_text, 'assistant': asst_text}

    def __len__(self): return len(self.samples)

    def __getitem__(self, idx):
        s = self.samples[idx]
        return {
            'image':     Image.open(s['som_path']).convert('RGB'),
            'user':      s['user'],
            'assistant': s['assistant'],
        }


class SoMCollator:
    def __init__(self, processor): self.processor = processor

    def __call__(self, batch):
        all_inputs = []
        for item in batch:
            messages = [
                {'role': 'system', 'content': SYSTEM_PROMPT},
                {'role': 'user', 'content': [
                    {'type': 'image', 'image': item['image']},
                    {'type': 'text',  'text':  item['user']},
                ]},
                {'role': 'assistant', 'content': item['assistant']},
            ]
            full_text   = self.processor.apply_chat_template(
                messages, tokenize=False, add_generation_prompt=False)
            prompt_text = self.processor.apply_chat_template(
                messages[:-1], tokenize=False, add_generation_prompt=True)
            image_inputs, _ = process_vision_info(messages)

            enc = self.processor(
                text=[full_text], images=image_inputs,
                padding='max_length', max_length=MAX_SEQ_LEN,
                truncation=True, return_tensors='pt')

            prompt_len = self.processor(
                text=[prompt_text], images=image_inputs,
                return_tensors='pt')['input_ids'].shape[1]

            labels = enc['input_ids'].clone()
            labels[0, :prompt_len] = -100
            labels[labels == self.processor.tokenizer.pad_token_id] = -100
            enc['labels'] = labels
            all_inputs.append(enc)

        return {k: torch.cat([x[k] for x in all_inputs], dim=0)
                for k in all_inputs[0]}

sample0 = dataset[0]

# ═══════════════════════════════════════════════════════════
# CELL 5c — Load model + apply LoRA
# ═══════════════════════════════════════════════════════════

print('Loading model in 4-bit...')
quant_config = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_compute_dtype=torch.float16,
    bnb_4bit_use_double_quant=True,
    bnb_4bit_quant_type='nf4',
)
model = _VLModel.from_pretrained(
    MODEL_ID,
    torch_dtype=torch.float16,
    device_map='auto',
    quantization_config=quant_config,
)
model.config.use_cache = False
model.enable_input_require_grads()

processor = AutoProcessor.from_pretrained(
    MODEL_ID,
    min_pixels=256 * 28 * 28,
    max_pixels=512 * 28 * 28,
)
print('Model loaded.')

print('Applying LoRA...')
lora_config = LoraConfig(
    task_type=TaskType.CAUSAL_LM,
    r=LORA_RANK,
    lora_alpha=LORA_ALPHA,
    lora_dropout=LORA_DROPOUT,
    target_modules=LORA_TARGETS,
    bias='none',
)
model = get_peft_model(model, lora_config)
model.print_trainable_parameters()


# ═══════════════════════════════════════════════════════════
# CELL 5d — Collator smoke test (1 sample, no training yet)
# Confirms image token count is fine before committing to training.
# ═══════════════════════════════════════════════════════════

collator = SoMCollator(processor)
test_batch = collator([dataset[0]])

print('Collator smoke test passed.')
print(f'  input_ids shape : {test_batch["input_ids"].shape}')
print(f'  labels shape    : {test_batch["labels"].shape}')
n_label_tokens = (test_batch['labels'] != -100).sum().item()
print(f'  label tokens    : {n_label_tokens}  (these are the assistant tokens we train on)')
assert n_label_tokens > 0, "No label tokens! Label masking is wrong."
print('Ready to train.')


# ═══════════════════════════════════════════════════════════
# CELL 5e — Train
# Expected time: ~4–6 hours on T4 for 3 epochs / 7k samples.
# Watch the first 3 log lines — loss should be dropping.
# ═══════════════════════════════════════════════════════════

training_args = SFTConfig(
    output_dir=OUTPUT_DIR,
    num_train_epochs=EPOCHS,
    per_device_train_batch_size=BATCH_SIZE,
    gradient_accumulation_steps=GRAD_ACCUM,
    learning_rate=LR,
    warmup_ratio=WARMUP_RATIO,
    lr_scheduler_type='cosine',
    fp16=True,
    gradient_checkpointing=True,
    logging_steps=10,
    save_steps=SAVE_STEPS,
    save_total_limit=2,
    remove_unused_columns=False,
    dataloader_num_workers=2,
    report_to='none',
    label_names=['labels'],
    max_length=MAX_SEQ_LEN,
    dataset_text_field=None,
)

trainer = SFTTrainer(
    model=model,
    args=training_args,
    train_dataset=dataset,
    data_collator=collator,
)

print('Starting training...')
print('Watch first few steps — loss should drop from ~2.5 toward ~0.5\n')
trainer.train()
print('\nTraining complete!')
